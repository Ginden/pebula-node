#Single Index

import { GtDocument, GtModel } from '@pebula/goosetyped';
@GtDocument()
export class Person extends GtModel() {
@GtIndex({ sort: 'asc' })
@GtColumn()
name: string;
@GtIndex({ sort: 'desc' })
@GtColumn()
age: number;
@GtIndex() // defaults to 'asc'
@GtColumn()
nickname: string;

}
